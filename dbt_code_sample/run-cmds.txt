mkdir dbt-duckdb-setup
cd dbt-duckdb-setup
uv venv
source .venv/bin/activate

uv pip install dbt-core dbt-duckdb

dbt init <dbt_code>

duckdb dev.duckdb

// with UI
duckdb -ui dev.duckdb

<WIN - ctrl + enter> to run the macro
<Mac - Command + enter> to run the macro
